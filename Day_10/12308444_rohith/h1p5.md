https://leetcode.com/problems/majority-element/

'''python

class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        count = {}

        for num in nums:
            count[num] = count.get(num, 0) + 1

        for key, value in count.items():
            if value > len(nums) // 2:
                return key
'''
